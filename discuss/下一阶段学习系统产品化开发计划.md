创建 `src/reflexlearn/learning/today.py`：

```python
from __future__ import annotations

from typing import Literal

from pydantic import BaseModel, Field


ResourceKind = Literal[
    "external_video",
    "ai_document",
    "quiz",
    "official_doc",
    "oer",
    "user_upload",
]

PathStatus = Literal["done", "current", "next"]


class TodayTask(BaseModel):
    title: str
    reason: str
    estimated_minutes: int = 30
    space_id: str = ""
    space_name: str = ""
    path_node: str = ""
    primary_action: str = "开始今日学习"


class TodayResource(BaseModel):
    id: str
    type: ResourceKind
    title: str
    provider: str
    source_label: str
    estimated_minutes: int
    reason: str
    href: str
    embed_url: str = ""
    usage_mode: str = "personal"
    source_policy: str = "owned_or_generated"


class TodayLearningPathNode(BaseModel):
    id: str
    title: str
    status: PathStatus
    summary: str


class ProfileSignal(BaseModel):
    label: str
    value: str


class TodayReviewItem(BaseModel):
    topic: str
    reason: str
    due_label: str


class TodaySummary(BaseModel):
    user_id: str
    greeting: str = "继续推进你的学习主线"
    current_goal: str = ""
    progress: float = 0.0
    main_task: TodayTask
    path_nodes: list[TodayLearningPathNode] = Field(default_factory=list)
    path_recommendation: str = ""
    resources: list[TodayResource] = Field(default_factory=list)
    tutor_prompt: str = "我可以用几个问题判断你是否真的理解当前知识点。"
    quick_actions: list[ProfileSignal] = Field(default_factory=list)
    profile_signals: list[ProfileSignal] = Field(default_factory=list)
    review_queue: list[TodayReviewItem] = Field(default_factory=list)
    degraded: list[str] = Field(default_factory=list)


def build_today_summary(
    *,
    user_id: str,
    spaces: list[dict],
    resources: list[dict],
    mistakes: list[dict],
    profile: dict,
    degraded: list[str] | None = None,
) -> TodaySummary:
    active_space = spaces[0] if spaces else {}
    goal = str(profile.get("goal") or active_space.get("title") or "创建你的第一个学习目标")
    weak_points = [str(x) for x in profile.get("weak_points", []) if str(x).strip()]
    first_weak = weak_points[0] if weak_points else "当前核心概念"
    second_weak = weak_points[1] if len(weak_points) > 1 else "下一步练习"
    progress = _float(profile.get("progress"), 0.0)

    main_task = TodayTask(
        title=f"先补齐{first_weak}，再进入{second_weak}",
        reason=f"系统根据你的画像和错题记录，建议先处理“{first_weak}”，再进入针对性练习。",
        estimated_minutes=32,
        space_id=str(active_space.get("space_id", "")),
        space_name=str(active_space.get("title", goal)),
        path_node=first_weak,
    )

    mapped_resources = [_resource_from_dict(item) for item in resources[:4]]
    if not mapped_resources:
        mapped_resources = _fallback_resources(first_weak)

    review_queue = [
        TodayReviewItem(
            topic=str(item.get("concept") or "待复习知识点"),
            reason="这类错题会影响后续路径推进，建议今天完成一次短复盘。",
            due_label="今天" if idx == 0 else "本周",
        )
        for idx, item in enumerate(mistakes[:3])
    ]

    preferences = profile.get("preferences", {})
    resource_mix = ""
    if isinstance(preferences, dict):
        resource_mix = str(preferences.get("resource_mix", "视频讲解 + 五题短练习"))

    profile_signals = [
        ProfileSignal(label="学习偏好", value=resource_mix or "视频讲解 + 五题短练习"),
        ProfileSignal(label="当前薄弱点", value="、".join(weak_points[:3]) or first_weak),
        ProfileSignal(label="学习目标", value=goal),
        ProfileSignal(label="完成进度", value=f"{round(progress * 100)}%"),
    ]

    return TodaySummary(
        user_id=user_id,
        current_goal=goal,
        progress=progress,
        main_task=main_task,
        path_nodes=[
            TodayLearningPathNode(
                id="foundation",
                title="建立直觉",
                status="done",
                summary="完成概念背景、问题形式和基础例子。",
            ),
            TodayLearningPathNode(
                id="current",
                title=first_weak,
                status="current",
                summary=f"正在处理“{first_weak}”相关卡点。",
            ),
            TodayLearningPathNode(
                id="next",
                title=second_weak,
                status="next",
                summary="下一步进入针对性练习和迁移应用。",
            ),
        ],
        path_recommendation=f"先处理“{first_weak}”可以降低后续学习负担。",
        resources=mapped_resources,
        quick_actions=[
            ProfileSignal(label="上传课程资料", value="/knowledge"),
            ProfileSignal(label="录入一道错题", value="/mistakes"),
            ProfileSignal(label="生成一组练习", value="/chat"),
            ProfileSignal(label="创建学习目标", value="/spaces"),
        ],
        profile_signals=profile_signals,
        review_queue=review_queue,
        degraded=degraded or [],
    )


def _resource_from_dict(item: dict) -> TodayResource:
    kind = str(item.get("type") or "ai_document")
    if kind not in {"external_video", "ai_document", "quiz", "official_doc", "oer", "user_upload"}:
        kind = _map_legacy_type(kind)
    return TodayResource(
        id=str(item.get("resource_id") or item.get("id") or item.get("title") or kind),
        type=kind,
        title=str(item.get("title") or item.get("type") or "学习资源"),
        provider=str(item.get("provider") or "ReflexLearn"),
        source_label=str(item.get("source_label") or _source_label(kind)),
        estimated_minutes=int(item.get("estimated_minutes") or 10),
        reason=str(item.get("reason") or "与当前学习目标相关。"),
        href=str(item.get("href") or "/resources"),
        embed_url=str(item.get("embed_url") or ""),
        usage_mode=str(item.get("usage_mode") or "personal"),
        source_policy=str(item.get("source_policy") or "owned_or_generated"),
    )


def _fallback_resources(concept: str) -> list[TodayResource]:
    return [
        TodayResource(
            id="fallback-bilibili-search",
            type="external_video",
            title=f"{concept}视频讲解",
            provider="Bilibili",
            source_label="B 站视频",
            estimated_minutes=14,
            reason="先用中文视频建立直觉，再回到讲义和练习。",
            href=f"https://search.bilibili.com/all?keyword={concept}",
            usage_mode="metadata_only",
            source_policy="embed_or_redirect_only",
        )
    ]


def _map_legacy_type(kind: str) -> ResourceKind:
    if kind == "doc":
        return "ai_document"
    if kind == "video":
        return "external_video"
    if kind == "reading":
        return "oer"
    if kind == "quiz":
        return "quiz"
    return "ai_document"


def _source_label(kind: str) -> str:
    labels = {
        "external_video": "外部视频",
        "ai_document": "AI 讲解文档",
        "quiz": "针对练习",
        "official_doc": "官方资料",
        "oer": "开放课程",
        "user_upload": "个人资料",
    }
    return labels.get(kind, "学习资源")


def _float(value: object, default: float) -> float:
    try:
        return float(value)
    except (TypeError, ValueError):
        return default
```

创建 `src/reflexlearn/api/routes/today.py`：

```python
from __future__ import annotations

from fastapi import APIRouter, Depends

from reflexlearn.api.deps import get_current_user
from reflexlearn.api.service_deps import safe_pg_pool
from reflexlearn.common.auth import CurrentUser
from reflexlearn.learning.assets import LearningAssetStore
from reflexlearn.learning.mistakes import MistakeStore
from reflexlearn.learning.today import TodaySummary, build_today_summary
from reflexlearn.memory import session_store

router = APIRouter()
_asset_store = LearningAssetStore()
_mistake_store = MistakeStore()


@router.get("/today")
async def get_today(user: CurrentUser = Depends(get_current_user)) -> TodaySummary:
    pg_pool = await safe_pg_pool()
    degraded: list[str] = []

    spaces_result = await _asset_store.list_spaces(
        user_id=user.user_id,
        tenant_id=user.tenant_id,
        pg_pool=pg_pool,
    )
    resources_result = await _asset_store.list_resources(
        user_id=user.user_id,
        tenant_id=user.tenant_id,
        pg_pool=pg_pool,
    )
    mistakes_result = await _mistake_store.list_for_user(
        user_id=user.user_id,
        tenant_id=user.tenant_id,
        pg_pool=pg_pool,
    )
    profile = await session_store.load_profile(user.user_id, tenant_id=user.tenant_id) or {}

    degraded.extend(getattr(spaces_result, "degraded", []))
    degraded.extend(getattr(resources_result, "degraded", []))

    return build_today_summary(
        user_id=user.user_id,
        spaces=[_dump(item) for item in spaces_result.items],
        resources=[_dump(item) for item in resources_result.items],
        mistakes=[_dump(item) for item in mistakes_result.items],
        profile=profile,
        degraded=degraded,
    )


def _dump(item: object) -> dict:
    if hasattr(item, "model_dump"):
        return item.model_dump()
    if isinstance(item, dict):
        return item
    return {}
```
