## 2. Docker Compose 配置

### 2.1 服务分组（profiles）

```yaml
# docker-compose.yml
services:
  # ============ core profile（日常开发必起）============
  api:
    build:
      context: .
      dockerfile: deploy/Dockerfile.api
    ports: ["8000:8000"]
    env_file: .env
    depends_on: [postgres, redis, qdrant]
    profiles: [core]
    volumes:
      - ./src:/app/src
      - ./config:/app/config
    command: uvicorn reflexlearn.main:app --host 0.0.0.0 --port 8000 --reload
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/api/health"]
      interval: 30s
      timeout: 5s
      retries: 3

  frontend:
    build:
      context: ./frontend
      dockerfile: ../deploy/Dockerfile.frontend
    ports: ["3000:3000"]
    depends_on: [api]
    profiles: [core]
    volumes:
      - ./frontend/src:/app/src
    command: npm run dev

  postgres:
    image: postgres:16-alpine
    ports: ["5432:5432"]
    environment:
      POSTGRES_DB: reflexlearn
      POSTGRES_USER: ${POSTGRES_USER:-reflexlearn}
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD:-reflexlearn}
    volumes:
      - pg_data:/var/lib/postgresql/data
    profiles: [core]
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U reflexlearn"]
      interval: 10s

  redis:
    image: redis:7-alpine
    ports: ["6379:6379"]
    volumes:
      - redis_data:/data
    profiles: [core]
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s

  qdrant:
    image: qdrant/qdrant:latest
    ports: ["6333:6333"]
    volumes:
      - qdrant_data:/qdrant/storage
    profiles: [core]
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:6333/healthz"]
      interval: 10s

  llm-gateway:
    image: ghcr.io/berriai/litellm:main-latest
    ports: ["4000:4000"]
    env_file: .env
    volumes:
      - ./config/llm_providers.yaml:/app/config.yaml
    command: --config /app/config.yaml
    profiles: [core]

  # ============ graph profile ============
  neo4j:
    image: neo4j:5-community
    ports: ["7474:7474", "7687:7687"]
    environment:
      NEO4J_AUTH: neo4j/${NEO4J_PASSWORD:-reflexlearn}
    volumes:
      - neo4j_data:/data
    profiles: [graph]

  # ============ bigdata profile ============
  kafka:
    image: bitnami/kafka:latest
    ports: ["9092:9092"]
    environment:
      KAFKA_CFG_NODE_ID: 0
      KAFKA_CFG_PROCESS_ROLES: broker,controller
      KAFKA_CFG_CONTROLLER_QUORUM_VOTERS: 0@kafka:9093
      KAFKA_CFG_LISTENERS: PLAINTEXT://:9092,CONTROLLER://:9093
      KAFKA_CFG_ADVERTISED_LISTENERS: PLAINTEXT://kafka:9092
      KAFKA_CFG_CONTROLLER_LISTENER_NAMES: CONTROLLER
    profiles: [bigdata]

  spark-master:
    image: bitnami/spark:latest
    ports: ["8080:8080", "7077:7077"]
    environment:
      SPARK_MODE: master
    profiles: [bigdata]

  spark-worker:
    image: bitnami/spark:latest
    environment:
      SPARK_MODE: worker
      SPARK_MASTER_URL: spark://spark-master:7077
    depends_on: [spark-master]
    profiles: [bigdata]

  namenode:
    image: bde2020/hadoop-namenode:2.0.0-hadoop3.2.1-java8
    ports: ["9870:9870"]
    environment:
      CLUSTER_NAME: reflexlearn
    volumes:
      - hdfs_namenode:/hadoop/dfs/name
    profiles: [bigdata]

  datanode:
    image: bde2020/hadoop-datanode:2.0.0-hadoop3.2.1-java8
    depends_on: [namenode]
    volumes:
      - hdfs_datanode:/hadoop/dfs/data
    profiles: [bigdata]

  worker:
    build:
      context: .
      dockerfile: deploy/Dockerfile.worker
    env_file: .env
    depends_on: [redis, postgres]
    profiles: [worker]

  # ============ observe profile ============
  prometheus:
    image: prom/prometheus:latest
    ports: ["9090:9090"]
    volumes:
      - ./deploy/prometheus/prometheus.yml:/etc/prometheus/prometheus.yml

  grafana:
    image: grafana/grafana:latest

volumes:
  pg_data:
  redis_data:
  qdrant_data:
  neo4j_data:
  hdfs_namenode:
  hdfs_datanode:
```

### 2.2 启动命令

```makefile
# Makefile
.PHONY: dev dev-full bigdata observe test eval

# 日常开发（最小集）
dev:
	docker compose --profile core up -d

# 带图谱
dev-graph:
	docker compose --profile core --profile graph up -d

# 全栈
dev-full:
	docker compose --profile core --profile graph --profile worker --profile observe up -d

# 大数据栈（按需）
bigdata:
	docker compose --profile bigdata up -d

# 可观测
observe:
	docker compose --profile observe up -d

# GPU 推理
gpu:
	docker compose --profile gpu up -d

# 停止所有
down:
	docker compose --profile core --profile graph --profile bigdata --profile worker --profile observe --profile gpu down

# 初始化数据库
init-db:
	python scripts/init_db.py && python scripts/init_qdrant.py && python scripts/init_neo4j.py

# 灌入种子数据
seed:
	python scripts/seed_data.py

# 运行测试
test:
	pytest tests/unit -v

test-integration:
	pytest tests/integration -v

# 运行评测
eval:
	python scripts/run_eval.py
```

```python
# 骨架：src/reflexlearn/common/config.py
from pydantic_settings import BaseSettings
from functools import lru_cache

class Settings(BaseSettings):
    # 数据库
    database_url: str
    redis_url: str = "redis://localhost:6379/0"
    qdrant_url: str = "http://localhost:6333"
    neo4j_uri: str = "bolt://localhost:7687"
    neo4j_password: str = "reflexlearn"

    # LLM
    deepseek_api_key: str = ""
    spark_api_key: str = ""
    qwen_api_key: str = ""
    anthropic_api_key: str = ""

    # Kafka
    kafka_bootstrap_servers: str = "localhost:9092"

    # 功能开关
    enable_reflexion: bool = True
    enable_rerank: bool = True
    enable_graph_retrieval: bool = True
    enable_kafka: bool = False
    enable_spark: bool = False

    # Harness
    max_iterations: int = 20
    max_replan: int = 3
    token_budget: int = 50000
    max_react_steps: int = 5

    class Config:
        env_file = ".env"

@lru_cache
def get_settings() -> Settings:
    return Settings()
```
